128 def create_output_file(type, input, residues, output, out_log):
129 # parse PDB file and get residues line by line
130 new_file_lines = []
131 curr_model = 0
132 with open(input) as infile:
133 for line in infile:
134 if line.startswith("MODEL "):
135 curr_model = line.rstrip()[-1]
136 if int(curr_model) > 1:
137 new_file_lines.append("ENDMDL\n")
138 new_file_lines.append("MODEL " + "{:>4}".format(curr_model) + "\n")
139  
140 conditional_atoms = [
141 (line.startswith("ATOM") or line.startswith("HETATM")),
142 line.startswith("HETATM"),
143 line.startswith("ATOM"),
144 ]
145  
146 if conditional_atoms[type]:
147 name = line[17:20].strip()
148 chain = line[21:22].strip()
149 res_id = line[22:27].strip()
150 if curr_model != 0:
151 model = curr_model.strip()
152 else:
153 model = "1"
154 if chain == "":
155 chain = " "
156  
157 for nstr in residues:
158 if (
159 nstr["res_id"] == res_id
  • W503 Line break before binary operator
160 and nstr["name"] == name
  • W503 Line break before binary operator
161 and nstr["chain"] == chain
  • W503 Line break before binary operator
162 and nstr["model"] == model
163 ):
164 new_file_lines.append(line)
165  
166 if int(curr_model) > 0:
167 new_file_lines.append("ENDMDL\n")
168  
169 fu.log("Writting pdb to: %s" % (output), out_log)
170  
171 # save new file with heteroatoms
172 with open(output, "w") as outfile:
173 for line in new_file_lines:
174 outfile.write(line)

241 def _from_string_to_list(input_data: Optional[Union[str, list[str]]]) -> list[str]:
242 """
243 Converts a string to a list, splitting by commas or spaces. If the input is already a list, returns it as is.
244 Returns an empty list if input_data is None.
245  
246 Parameters:
247 input_data (str, list, or None): The string, list, or None value to convert.
248  
249 Returns:
250 list: A list of string elements or an empty list if input_data is None.
251 """
252 if input_data is None:
253 return []
254  
255 if isinstance(input_data, list):
256 # If input is already a list, return it
257 return input_data
258  
259 # If input is a string, determine the delimiter based on presence of commas
260 delimiter = "," if "," in input_data else " "
261 items = input_data.split(delimiter)
262  
263 # Remove whitespace from each item and ignore empty strings
264 processed_items = [item.strip() for item in items if item.strip()]
265  
266 return processed_items
